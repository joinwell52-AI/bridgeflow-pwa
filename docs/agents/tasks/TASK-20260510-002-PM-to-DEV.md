## §三 P1：真 Cursor SDK 接入 + 配置加载（1 工作日）

### P1 §1 修改文件

#### `codeflow-shell/src/sdk-factory.ts`

```typescript
// 删掉 return null 改为：
import { CursorSdkAdapter } from "@codeflow/runtime";

export function makeRealCursorSdkAdapter(
  cfg: { apiKey?: string; defaultModel?: string; listScope?: "local" | "cloud" }
): AgentSdkAdapter | null {
  const apiKey = cfg.apiKey ?? process.env["CURSOR_API_KEY"];
  if (!apiKey) return null;  // 仍 fallback to fake adapter
  return new CursorSdkAdapter({
    apiKey,
    listScope: cfg.listScope ?? "local",
  });
}
```
