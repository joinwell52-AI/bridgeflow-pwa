## 必交付（1 项）

在 `packages/codeflow-runtime/src/registry/__tests__/PersistentStore.test.ts` 末尾追加 1 个测试：

```ts
test("scenario 11: concurrent upsert via Promise.all does not corrupt agents.json", async () => {
  // arrange: empty store
  const dir = await mkTempDir();
  const store = new PersistentStore({ dir });

  // act: 5 concurrent upserts with different agent ids
  const records: AgentRecord[] = Array.from({ length: 5 }, (_, i) => ({
    agent_id: `dev-${i.toString().padStart(2, "0")}`,
    sdk_agent_id: `sdk-${i}`,
    layer: "worker",
    runtime: "cursor-sdk@1.0.12",
    status: "idle",
    created_at: new Date().toISOString(),
    reconciled_at: new Date().toISOString(),
  }));

  await Promise.all(records.map((r) => store.upsert(r)));

  // assert: file readable as valid JSON
  const loaded = await store.loadAll();

  // 不要求 5 个全在——atomic rename 序列化下，
  // 后到的写入会覆盖前面的，但每一次都必须是 *valid* JSON
  // 关键 invariant：loadAll 不抛 RegistryWriteError，文件不是半残
  assert.ok(Array.isArray(loaded));
  assert.ok(loaded.length >= 1, "at least one upsert survived");
  assert.ok(loaded.length <= 5, "no extra ghost records");

  // 全部 loaded 中的 record 都是结构合法的（schema 通过）
  for (const r of loaded) {
    assert.ok(r.agent_id);
    assert.ok(r.sdk_agent_id);
    assert.ok(r.layer === "worker");
  }
});
```

**关键 invariant**：测的不是"5 次 upsert 全部成功"——**atomic-rename + 无锁 = 后写覆盖前写**是 PersistentStore 当前设计的行为（不引入锁是 Phase A 决策 D）。我们要测的是**"即使后写覆盖前写，agents.json 永远不会半残"**——`loadAll()` 不抛、不读到 partial JSON。
