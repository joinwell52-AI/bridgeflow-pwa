### §3.1 `CursorSdkAdapterOptions` 加 `defaultModel` 字段

文件：`packages/codeflow-runtime/src/registry/AgentSdkAdapter.ts`

```typescript
export interface CursorSdkAdapterOptions {
  apiKey?: string;
  defaultCwd?: string;
  listScope?: "local" | "cloud" | undefined;
  /** Default model id forwarded to Agent.create / agent.send when caller
   *  does not specify one. Required for `local` runtime mode (Cursor SDK
   *  rejects local agents without explicit model). MT-1 / BUG-SDK-001. */
  defaultModel?: string;          // ← 新增
}
```

### §3.2 `CursorSdkAdapter.create()` + `.send()` 注入 model

```typescript
async create(spec: AgentCreateSpec): Promise<{ sdk_agent_id: string }> {
  const apiKey = this._resolveApiKey();
  const modelId = spec.modelId ?? this._opts.defaultModel;
  let agent;
  try {
    agent = await Agent.create({
      apiKey,
      name: `CodeFlow ${spec.agentId}`,
      ...(modelId ? { model: { id: modelId } } : {}),  // ← 改：spec.modelId ?? defaultModel
      local: { cwd: spec.workspace ?? this._opts.defaultCwd ?? process.cwd() },
    });
    // ...
  }
}

async send(spec: AgentSendSpec, sdkAgentId: string): Promise<RunHandle> {
  const apiKey = this._resolveApiKey();
  const modelId = spec.modelId ?? this._opts.defaultModel;
  let agent;
  try {
    agent = await Agent.resume(sdkAgentId, {
      apiKey,
      ...(modelId ? { model: { id: modelId } } : {}),  // ← 改
      local: { cwd: this._opts.defaultCwd ?? process.cwd() },
    });
    // ...
  }
}
```

### §3.3 `sdk-factory.ts` 把 cfg.cursor.defaultModel 传进去

文件：`codeflow-shell/src/sdk-factory.ts`

```typescript
export function makeRealCursorSdkAdapter(
  cfg: { apiKey?: string; defaultModel?: string; listScope: "local" | "cloud" },
): AgentSdkAdapter | null {
  if (!cfg.apiKey) return null;
  return new CursorSdkAdapter({
    apiKey: cfg.apiKey,
    listScope: cfg.listScope,
    defaultModel: cfg.defaultModel,  // ← 新增
  });
}
```
